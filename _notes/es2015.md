## Arrow Functions
```javascript
// automatically bind the lexical `this`
fetch('www...').then((response) => {
  this.name = response.body.name;
});

// sames as
fetch('www...').then(function(response) => {
  this.name = response.body.name;
}.bind(this));

// wrap statement in () or omit () 
// same as returning statement
fetch('www...').then((response) => (
  response.body.name;
));

fetch('www...').then((response) => response.body.name);

// sames as
fetch('www...').then((response) => {
  return response.body.name;
});

// destructuring
fetch('www...').then(({ body: { name }}) => name);
```

## Destructing

```javascript
// Array matching
const list = [1, 2, 3]
const [a, b, c] = list
a -> 1
b -> 2
c -> 3

// ...spread operator
const list = [1, 2, 3, 4, 5]
const [a, b, ...c] = list
a -> 1
b -> 2
c -> [3, 4, 5]

// skip values
const list = [1, 2, 3]
const [a, ,c] = list
a -> 1
c -> 3

// Oject matching
const obj = { r: 1, p: 2, q: 3 }
const { a, b, c } = obj
a -> 1
b -> 2
c -> 3

const obj = { r: { p: 2, q: { l: 3 } } }
const { a { b, c { d } } } = obj
b -> 2
d -> 3

// parameter context matching
function log({ value }) { console.log(value) }
log({ value: "Hey man!" });

// can pass non-object var named value
const value = "go hard"
log(value)
-> go hard

```
